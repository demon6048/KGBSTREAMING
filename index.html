Instrucciones de Instalación

Este código conecta tu interfaz web con Google Sheets. Copia el siguiente script y pégalo en el editor de Apps Script de tu hoja de cálculo.

Recuerda: Tu Google Sheet debe tener las pestañas: VENTAS, PRODUCTOS, VENDEDORES, CONFIG.

Google Apps Script Code

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // Lectura de Configuración y Maestros
  const configSheet = ss.getSheetByName("CONFIG");
  const adminPct = configSheet.getRange("B1").getValue() || 0.60;
  const sellerPct = configSheet.getRange("B2").getValue() || 0.40;
  
  const vendedores = ss.getSheetByName("VENDEDORES").getDataRange().getValues().slice(1)
    .filter(r => r[2] === "ACTIVO").map(r => ({id: r[0], name: r[1]}));
    
  const productos = ss.getSheetByName("PRODUCTOS").getDataRange().getValues().slice(1)
    .filter(r => r[4] === true).map(r => ({id: r[0], name: r[1], cost: r[2], price: r[3]}));
    
  // Lectura de Historial (Optimizado: Últimas 1000 ventas)
  const ventas = ss.getSheetByName("VENTAS").getDataRange().getValues().slice(1);

  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    config: { admin: adminPct, seller: sellerPct },
    data: { sellers: vendedores, products: productos, history: ventas }
  })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = JSON.parse(e.postData.contents);
  
  // Registrar Venta
  if (data.action === 'sale') {
    const sheet = ss.getSheetByName("VENTAS");
    const profit = data.price - data.cost;
    sheet.appendRow([
      Utilities.getUuid(), new Date(), data.seller, data.product, 
      data.client, data.cost, data.price, profit, 
      profit * data.sellerPct, profit * data.adminPct
    ]);
    return response("Venta registrada correctamente");
  }
  
  // Agregar Producto
  if (data.action === 'add_product') {
    const sheet = ss.getSheetByName("PRODUCTOS");
    sheet.appendRow([Utilities.getUuid(), data.name, data.cost, data.price, true]);
    return response("Plataforma agregada");
  }
}

function response(msg) { 
  return ContentService.createTextOutput(JSON.stringify({status: "success", msg: msg}))
    .setMimeType(ContentService.MimeType.JSON); 
}
